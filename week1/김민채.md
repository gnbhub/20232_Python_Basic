```py
water = 0
x= 200  
y = 50  
z = 1500  
days = 0 

while water < z:
    days += 1
    water += x - y
    print(f"day : {days}  세숫대야의 물 : {water} ml")

print(f"\n세숫대야가 넘치는 날은 {days}일 입니다.")
```
