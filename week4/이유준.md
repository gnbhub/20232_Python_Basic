```py
input_word = input()
alfabet = 'abcdefghijklmnopqrstuvwxyz'

for i in alfabet:
    if i in input_word:
        print(input_word.index(i), end=' ')
    else:
        print(-1, end=' ')
```

#2.
```py
i = int(input("총 몇명의 점수를 계산할 것인가? : "))
tscore = 0

for s in range(i):
    score = int(input("점수를 입력하시오 : "))
    tscore = tscore + score
    
print("총 점수는 %d 입니다.", tscore)
```
